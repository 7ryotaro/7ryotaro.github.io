var kentX=150;
var kentY=50;

void setup() {
  size(300, 300);
}

void draw() {
fill(108, 245, 22);
strokeWeight(8);
ellipse(kentX , kentY, 100, 100);

rect(kentX-50, kentY+61, 100, 200);


line(kentX-50, kentY+71, kentX-100, kentY+170);
line(kentX+50, kentY+77, kentX+100 ,kentY+170);
line(kentX-50, kentY+250, kentX-50, kentY+350);
line(kentX+50, kentY+250, kentX+50, kentY+350);

strokeWeight(15);
point(kentX-20, kentY-5);
point(kentX+20, kentY-5);
point(kentX, kentY+10);

strokeWeight(3);
line(kentX-20, kentY+25, kentX+20, kentY+25);

strokeWeight(5);
ellipse(kentX, kentY+110, 40, 39);
ellipse(kentX, kentY+162, 40, 40);
ellipse(kentX, kentY+213, 40, 38);
};
